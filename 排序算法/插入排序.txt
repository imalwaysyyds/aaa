#include <iostream>
#include <vector>

using namespace std;

// 插入排序函数，使用模板以便可以处理不同类型的数据
template<typename T>
void insertionSort(vector<T>& arr) {
    int n = arr.size();
    for (int i = 1; i < n; ++i) {
        T key = arr[i];
        int j = i - 1;

        // 将key与已排序部分的元素从后往前依次比较，并移动元素
        while (j >= 0 && arr[j] > key) {
            arr[j + 1] = arr[j];
            j--;
        }

        // 将key插入到合适的位置
        arr[j + 1] = key;
    }
}

int main() {
    // 示例用法，创建一个整型数组并调用插入排序函数
    vector<int> numbers = { 5, 3, 4, 6, 2 };
    insertionSort(numbers);

    // 输出排序后的数组
    for (int num : numbers) {
        cout << num << " ";
    }
    cout << endl;

    return 0;
}
