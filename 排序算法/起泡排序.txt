#include <iostream>
#include <vector>

using namespace std;

// 冒泡排序函数，使用模板可处理不同类型的数据
template<typename T>
void bubbleSort(vector<T>& arr) {
    int n = arr.size();
    for (int i = 0; i < n - 1; ++i) {
        for (int j = 0; j < n - i - 1; ++j) {
            if (arr[j] > arr[j + 1]) {
                // 交换元素位置
                T temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

int main() {
    // 示例用法，创建一个整型数组并调用冒泡排序函数
    vector<int> numbers = { 5, 3, 4, 6, 2 };
    bubbleSort(numbers);

    // 输出排序后的数组
    for (int num : numbers) {
        cout << num << " ";
    }
    cout << endl;

    return 0;
}
